public float speed;
	public bool canMove = true;
	public Vector2 originalPosition;
	public bool colidiu = false;
	// Use this for initialization
	void OnCollisionEnter2D(Collision2D coll) {
		if (coll.gameObject.tag == "Limit"  || coll.gameObject.tag == "Barco"){
			canMove = false;
			colidiu = true;
		}
	}
	void OnMouseOver (){
		if (Input.GetKey(KeyCode.Mouse0)){
			canMove = false;
			Vector2 pos = Camera.main.ScreenToWorldPoint(Input.mousePosition); 	
			transform.position = pos; 
			gameObject.GetComponent<Collider2D> ().isTrigger = true;
			
			GameObject[] objs = GameObject.FindGameObjectsWithTag("Barco");
			foreach (GameObject obj in objs){
				if (obj.GetComponent <Moviment> () != null)
					obj.GetComponent <Moviment> ().canMove = false;
			}
		}
		if (Input.GetKeyUp(KeyCode.Mouse0)){
			gameObject.GetComponent<Collider2D> ().isTrigger = false;
			if (!colidiu){
				BackOriginalPosition ();
				canMove = true;
			}
			GameObject[] objs = GameObject.FindGameObjectsWithTag("Barco");
			foreach (GameObject obj in objs){
				if (obj.GetComponent <Moviment> () != null)
					obj.GetComponent <Moviment> ().canMove = true;
			}
		}
	}
	
	public void BackOriginalPosition (){
		transform.position = originalPosition;
		canMove = true;
	}
	
	// Update is called once per frame
	void Update () {
		if (canMove){
			transform.Translate ( -speed * Time.deltaTime, 0f, 0f); 
			originalPosition = transform.position; 
		}
	}
